33  public interface TupleSpace extends JavaSpace05 {
34      
35      /**
36       * When entries are written that match this template notify the
37       * given <code>listener</code> with a <code>RemoteEvent</code> that
38       * includes the <code>handback</code> object.  Matching is done as
39       * for <code>read</code>.
40       *
41       * @param tmpl      The template used for matching.  Matching is
42       *			done against <code>tmpl</code> with <code>null</code>
43       *			fields being wildcards ("match anything") other
44       *			fields being values ("match exactly on the
45       *			serialized form").
46       * @param txn	The transaction (if any) under which to work.
47       * @param listener  The remote event listener to notify.
48       * @param lease  the requested lease time, in milliseconds
49       * @param handback  An object to send to the listener as part of the 
50       *                  event notification.
51       * @return the event registration to the the registrant
52       * @throws TransactionException if a transaction error occurs
53       * @throws RemoteException if a communication error occurs
54       * @throws IllegalArgumentException if the lease time requested 
55       *         is not Lease.ANY and is negative
56       * @see #read
57       * @see net.jini.core.event.EventRegistration
58       */
59      EventRegistration
60  	notify(Entry tmpl, Transaction txn, RemoteEventListener listener,
61  	       long lease, MarshalledInstance handback)
62  	throws TransactionException, RemoteException;
63      
64      /**
65       * Register for events triggered when a matching {@link Entry}
66       * transitions from unavailable to available.  The resulting
67       * events will be instances of the {@link AvailabilityEvent}
68       * class and the {@link AvailabilityEvent#getEntry
69       * AvailabilityEvent.getEntry} method will return a copy of
70       * the <code>Entry</code> whose transition triggered the
71       * event.<p>
72       *
73       * An <code>Entry</code> makes a transition from
74       * <em>unavailable to available</em> when it goes from being in
75       * a state where it could not be returned by a {@link
76       * TupleSpace#take TupleSpace.take} using <code>txn</code> to a
77       * state where it could be returned. An <code>Entry</code>
78       * makes a transition from <em>invisible to visible</em> when
79       * it goes from being in a state where it could not be returned
80       * by a {@link TupleSpace#read TupleSpace.read} using
81       * <code>txn</code> to a state where it could be
82       * returned. Note, any transition from invisible to visible is
83       * also a transition from unavailable to available, but an
84       * already visible entry can be unavailable and then make a
85       * transition from unavailable to available. Because the entry
86       * was already visible, this transition would not be a
87       * transition from invisible to visible.<p>
88       *
89       * The <code>tmpls</code> parameter must be a {@link
90       * Collection} of <code>Entry</code> instances to be used as
91       * templates. Events will be generated when an
92       * <code>Entry</code> that matches one or more of these
93       * templates makes an appropriate transition. A single
94       * transition will generate only one event per registration, in
95       * particular the transition of an <code>Entry</code> that
96       * matches multiple elements of <code>tmpls</code> must still
97       * generate exactly one event for this registration. If a given
98       * <code>Entry</code> undergoes multiple applicable transitions
99       * while the registration is active, each must generate a
100      * separate event.<p>
101      * 
102      * Events are not generated directly by the transition of
103      * matching entries, but instead by an abstract observer set up
104      * in the space for each registration. The observer may see the
105      * transitions out of order and as a result the order of the
106      * events generated for this registration (as determined by the
107      * sequence numbers assigned to the events) may be different
108      * from the order of the transitions themselves. Additionally,
109      * each registration will have its own abstract observer and
110      * different observers may see the same sequence of transitions
111      * in different orders. As a result, given a set of transitions
112      * that trigger events for two different registrations, the
113      * order of the events generated for one registration may
114      * differ from the order of the events generated for the
115      * other. <p>
116      *
117      * A non-<code>null</code> {@link EventRegistration} object
118      * will be returned.  Each registration will be assigned an
119      * event ID. The event ID will be unique at least with respect
120      * to all other active event registrations for
121      * <code>AvailabilityEvent</code>s on this space with a
122      * non-equivalent set of templates, a different transaction,
123      * and/or a different value for the
124      * <code>visibilityOnly</code> flag. The event ID can be
125      * obtained by calling the {@link EventRegistration#getID
126      * EventRegistration.getID} method on the returned
127      * <code>EventRegistration</code>. The returned
128      * <code>EventRegistration</code> object's {@link
129      * EventRegistration#getSource EventRegistration.getSource}
130      * method will return a reference to the space.<p>
131      * 
132      * Registrations are leased. <code>leaseDurations</code>
133      * represents the client's desired initial lease duration.  If
134      * <code>leaseDuration</code> is positive, the initial lease
135      * duration will be a positive value less than or equal to
136      * <code>leaseDuration</code>. If <code>leaseDuration</code> is
137      * {@link Lease#ANY Lease.ANY}, the space is free to pick any
138      * positive initial lease duration it desires. A proxy for the
139      * lease associated with the registration can be obtained by
140      * calling the returned <code>EventRegistration</code>'s {@link
141      * EventRegistration#getLease EventRegistration.getLease}
142      * method.<p>
143      *
144      * A registration made with a non-<code>null</code> value for
145      * <code>txn</code> is implicitly dropped when the space
146      * observes <code>txn</code> has left the active state.<p>
147      *
148      * @param tmpls a {@link Collection} of {@link Entry}
149      *              instances, each representing a
150      *              template. Events for this registration will be
151      *              generated by the transitions of entries
152      *              matching one or more elements of
153      *              <code>tmpls</code>
154      * @param txn   the {@link Transaction} this operation should be
155      *              performed under, may be <code>null</code>
156      * @param visibilityOnly if <code>true</code>, events will
157      *              be generated for this registration only when a
158      *              matching <code>Entry</code> transitions from
159      *              invisible to visible, otherwise events will be
160      *              generated when a matching <code>Entry</code>
161      *              makes any transition from unavailable to
162      *              available
163      * @param listener the object to which events generated for
164      *              this registration should be delivered
165      * @param leaseDuration the requested initial lease time on
166      *              the resulting event registration
167      * @param handback the {@link MarshalledInstance} to be 
168      *              returned by the {@link
169      *              RemoteEvent#getRegistrationObject 
170      *              RemoteEvent.getRegistrationObject} method of
171      *              the events generated for this registration
172      * @return an {@link EventRegistration} object with
173      *         information on this registration
174      * @throws TransactionException if <code>txn</code> is
175      *         non-<code>null</code> and is not usable by the
176      *         space
177      * @throws RemoteException if a communication error occurs
178      * @throws IllegalArgumentException if any non-<code>null</code>
179      *         element of <code>tmpls</code> is not an instance of
180      *         <code>Entry</code>, if <code>tmpls</code> is empty,
181      *         or if <code>leaseDuration</code> is neither
182      *         positive nor {@link Lease#ANY Lease.ANY}
183      * @throws NullPointerException if <code>tmpls</code> or
184      *         <code>listener</code> is <code>null</code> 
185      */
186     public EventRegistration 
187 	registerForAvailabilityEvent(Collection          tmpls, 
188 				     Transaction         txn,
189 				     boolean             visibilityOnly,
190 				     RemoteEventListener listener,
191 				     long                leaseDuration, 
192 				     MarshalledInstance  handback)
193 	throws TransactionException, RemoteException;
194     
195 }
